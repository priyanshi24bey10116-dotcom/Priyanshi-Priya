SELECT * 
FROM departments 
LEFT OUTER JOIN hod 
ON (departments.did = hod.hid) 
ORDER BY did ASC;

SELECT * 
FROM departments 
INNER JOIN hod 
ON departments.hid = hod.hid;

SELECT * 
FROM departments 
CROSS JOIN hod 
ORDER BY did;

SELECT * 
FROM departments NATURAL JOIN hod;
